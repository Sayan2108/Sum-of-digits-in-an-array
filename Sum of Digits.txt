import java.util.*;

class practise{
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] array = new int[n];
        for(int i=0;i<n;i++){
            array[i] = sc.nextInt();
        }
        int sum = 0;
        for(int i=0;i<n;i++){
        sum += array[i];    
        }
        System.out.println(sum);
    }
}
